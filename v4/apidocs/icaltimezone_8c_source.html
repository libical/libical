392char *icaltimezone_get_tznames_from_vtimezone(icalcomponent *component)
393{
394 icalcomponent *comp;
395 struct icaltimetype dtstart;
396 struct icaldatetimeperiodtype rdate;
397 const char *current_tzname;
398 const char *standard_tzname = NULL, *daylight_tzname = NULL;
399 struct icaltimetype standard_max_date, daylight_max_date;
400
401 standard_max_date = icaltime_null_time();
402 daylight_max_date = icaltime_null_time();
403
404 /* Step through the STANDARD & DAYLIGHT subcomponents. */
405 comp = icalcomponent_get_first_component(component, ICAL_ANY_COMPONENT);
406 while (comp) {
407 icalcomponent_kind type = icalcomponent_isa(comp);
408 if (type == ICAL_XSTANDARD_COMPONENT || type == ICAL_XDAYLIGHT_COMPONENT) {
409 struct icaltimetype current_max_date = icaltime_null_time();
410 current_tzname = NULL;
411
412 /* Step through the properties. We want to find the TZNAME, and
413 the largest DTSTART or RDATE. */
414 icalproperty *prop = icalcomponent_get_first_property(comp, ICAL_ANY_PROPERTY);
415 while (prop) {
416 switch (icalproperty_isa(prop)) {
417 case ICAL_TZNAME_PROPERTY:
418 current_tzname = icalproperty_get_tzname(prop);
419 break;
420
421 case ICAL_DTSTART_PROPERTY:
422 dtstart = icalproperty_get_dtstart(prop);
423 if (icaltime_compare(dtstart, current_max_date) > 0) {
424 current_max_date = dtstart;
425 }
426
427 break;
428
429 case ICAL_RDATE_PROPERTY:
430 rdate = icalproperty_get_rdate(prop);
431 if (icaltime_compare(rdate.time, current_max_date) > 0) {
432 current_max_date = rdate.time;
433 }
434
435 break;
436
437 default:
438 break;
439 }
440
441 prop = icalcomponent_get_next_property(comp, ICAL_ANY_PROPERTY);
442 }
443
444 if (current_tzname) {
445 if (type == ICAL_XSTANDARD_COMPONENT) {
446 if (!standard_tzname ||
447 icaltime_compare(current_max_date, standard_max_date) > 0) {
448 standard_max_date = current_max_date;
449 standard_tzname = current_tzname;
450 }
451 } else {
452 if (!daylight_tzname ||
453 icaltime_compare(current_max_date, daylight_max_date) > 0) {
454 daylight_max_date = current_max_date;
455 daylight_tzname = current_tzname;
456 }
457 }
458 }
459 }
460
461 comp = icalcomponent_get_next_component(component, ICAL_ANY_COMPONENT);
462 }
463
464 /* Outlook (2000) places "Standard Time" and "Daylight Time" in the TZNAME
465 strings, which is totally useless. So we return NULL in that case. */
466 if (standard_tzname && !strcmp(standard_tzname, "Standard Time")) {
467 return NULL;
468 }
469
470 /* If both standard and daylight TZNAMEs were found, if they are the same
471 we return just one, else we format them like "EST/EDT". */
472 if (standard_tzname && daylight_tzname) {
473 size_t standard_len, daylight_len;
474 char *tznames;
475
476 if (!strcmp(standard_tzname, daylight_tzname)) {
477 return icalmemory_strdup(standard_tzname);
478 }
479
480 standard_len = strlen(standard_tzname);
481 daylight_len = strlen(daylight_tzname);
482 const size_t len_tznames = standard_len + daylight_len + 2;
483 tznames = icalmemory_new_buffer(len_tznames);
484 strncpy(tznames, standard_tzname, len_tznames);
485 tznames[standard_len] = '/';
486 strncpy(tznames + standard_len + 1, daylight_tzname, len_tznames - standard_len - 1);
487 tznames[len_tznames - 1] = '\0';
488 return tznames;
489 } else {
490 const char *tznames;
491
492 /* If either of the TZNAMEs was found just return that, else NULL. */
493 tznames = standard_tzname ? standard_tzname : daylight_tzname;
494 return tznames ? icalmemory_strdup(tznames) : NULL;
495 }
496}

498static void icaltimezone_ensure_coverage(icaltimezone *zone, int end_year)
499{
500 /* When we expand timezone changes we always expand at least up to this
501 year, plus ICALTIMEZONE_EXTRA_COVERAGE. */
502 static ICAL_GLOBAL_VAR int icaltimezone_minimum_expansion_year = -1;
503
504 if (!zone) {
505 return;
506 }
507
508 icaltimezone_load_builtin_timezone(zone);
509
510 if (icaltimezone_minimum_expansion_year == -1) {
511 struct icaltimetype today = icaltime_today();
512
513 icaltimezone_minimum_expansion_year = today.year;
514 }
515
516 int changes_end_year = end_year;
517 if (changes_end_year < icaltimezone_minimum_expansion_year) {
518 changes_end_year = icaltimezone_minimum_expansion_year;
519 }
520
521 changes_end_year += ICALTIMEZONE_EXTRA_COVERAGE;
522
523 if (changes_end_year > ICALTIMEZONE_MAX_YEAR) {
524 changes_end_year = ICALTIMEZONE_MAX_YEAR;
525 }
526
527 if (!zone->changes || zone->end_year < end_year) {
528 icaltimezone_expand_changes(zone, changes_end_year);
529 }
530}

532/* Hold the icaltimezone_changes_lock(); before calling this function */
533static void icaltimezone_expand_changes(icaltimezone *zone, int end_year)
534{
535 icalarray *changes;
536 icalcomponent *comp;
537
538 if (!zone) {
539 return;
540 }
541#ifdef ICALTIMEZONE_DEBUG_PRINT
542 printf("\nExpanding changes for: %s to year: %i\n", zone->tzid, end_year);
543#endif
544
545 changes = icalarray_new(sizeof(icaltimezonechange), 32);
546 if (!changes) {
547 return;
548 }
549
550 /* Scan the STANDARD and DAYLIGHT subcomponents. */
551 comp = icalcomponent_get_first_component(zone->component, ICAL_ANY_COMPONENT);
552 while (comp) {
553 icaltimezone_expand_vtimezone(comp, end_year, changes);
554 comp = icalcomponent_get_next_component(zone->component, ICAL_ANY_COMPONENT);
555 }
556
557 /* Sort the changes. We may have duplicates but I don't think it will
558 matter. */
559 icalarray_sort(changes, icaltimezone_compare_change_fn);
560
561 if (zone->changes) {
562 icalarray_free(zone->changes);
563 zone->changes = 0;
564 }
565 zone->changes = changes;
566 zone->end_year = end_year;
567}

569void icaltimezone_expand_vtimezone(icalcomponent *comp, int end_year, icalarray *changes)
570{
571 icaltimezonechange change;
572 icalproperty *prop;
573 struct icaltimetype dtstart, occ;
574 struct icalrecurrencetype *rrule;
575 icalrecur_iterator *rrule_iterator;
576 struct icaldatetimeperiodtype rdate;
577 int found_dtstart = 0, found_tzoffsetto = 0, found_tzoffsetfrom = 0;
578 int has_rdate = 0, has_rrule = 0;
579
580 /* First we check if it is a STANDARD or DAYLIGHT component, and
581 just return if it isn't. */
582 if (icalcomponent_isa(comp) == ICAL_XSTANDARD_COMPONENT) {
583 change.is_daylight = 0;
584 } else if (icalcomponent_isa(comp) == ICAL_XDAYLIGHT_COMPONENT) {
585 change.is_daylight = 1;
586 } else {
587 return;
588 }
589
590 /* Step through each of the properties to find the DTSTART,
591 TZOFFSETFROM and TZOFFSETTO. We can't expand recurrences here
592 since we need these properties before we can do that. */
593 prop = icalcomponent_get_first_property(comp, ICAL_ANY_PROPERTY);
594 while (prop) {
595 switch (icalproperty_isa(prop)) {
596 case ICAL_DTSTART_PROPERTY:
597 dtstart = icalproperty_get_dtstart(prop);
598 found_dtstart = 1;
599 break;
600 case ICAL_TZOFFSETTO_PROPERTY:
601 change.utc_offset = icalproperty_get_tzoffsetto(prop);
602 /*printf ("Found TZOFFSETTO: %i\n", change.utc_offset); */
603 found_tzoffsetto = 1;
604 break;
605 case ICAL_TZOFFSETFROM_PROPERTY:
606 change.prev_utc_offset = icalproperty_get_tzoffsetfrom(prop);
607 /*printf ("Found TZOFFSETFROM: %i\n", change.prev_utc_offset); */
608 found_tzoffsetfrom = 1;
609 break;
610 case ICAL_RDATE_PROPERTY:
611 has_rdate = 1;
612 break;
613 case ICAL_RRULE_PROPERTY:
614 has_rrule = 1;
615 break;
616 default:
617 /* Just ignore any other properties. */
618 break;
619 }
620
621 prop = icalcomponent_get_next_property(comp, ICAL_ANY_PROPERTY);
622 }
623
624 /* Microsoft Outlook for Mac (and possibly other versions) will create
625 timezones without a tzoffsetfrom property if it's a timezone that
626 doesn't change for DST. */
627 if (found_tzoffsetto && !found_tzoffsetfrom) {
628 change.prev_utc_offset = change.utc_offset;
629 found_tzoffsetfrom = 1;
630 }
631
632 /* If we didn't find a DTSTART, TZOFFSETTO and TZOFFSETFROM we have to
633 ignore the component. FIXME: Add an error property? */
634 if (!found_dtstart || !found_tzoffsetto || !found_tzoffsetfrom) {
635 return;
636 }
637
638#ifdef ICALTIMEZONE_DEBUG_PRINT
639 printf("\n Expanding component DTSTART (Y/M/D): %i/%i/%i %i:%02i:%02i\n",
640 dtstart.year, dtstart.month, dtstart.day, dtstart.hour, dtstart.minute, dtstart.second);
641#endif
642
643 /* If the STANDARD/DAYLIGHT component has no recurrence rule, we add
644 a single change for the DTSTART. */
645 if (!has_rrule) {
646 change.year = dtstart.year;
647 change.month = dtstart.month;
648 change.day = dtstart.day;
649 change.hour = dtstart.hour;
650 change.minute = dtstart.minute;
651 change.second = dtstart.second;
652
653 /* Convert to UTC. */
654 icaltimezone_adjust_change(&change, 0, 0, 0, -change.prev_utc_offset);
655
656#ifdef ICALTIMEZONE_DEBUG_PRINT
657 printf(" Appending single DTSTART (Y/M/D): %i/%02i/%02i %i:%02i:%02i\n",
658 change.year, change.month, change.day, change.hour, change.minute, change.second);
659#endif
660
661 /* Add the change to the array. */
662 icalarray_append(changes, &change);
663 }
664
665 const size_t max_rrule_search = icallimit_get(ICAL_LIMIT_RRULE_SEARCH);
666
667 /* The component has recurrence data, so we expand that now. */
668 prop = icalcomponent_get_first_property(comp, ICAL_ANY_PROPERTY);
669 while (prop && (has_rdate || has_rrule)) {
670#ifdef ICALTIMEZONE_DEBUG_PRINT
671 printf("Expanding property...\n");
672#endif
673 switch (icalproperty_isa(prop)) {
674 case ICAL_RDATE_PROPERTY:
675 rdate = icalproperty_get_rdate(prop);
676 change.year = rdate.time.year;
677 change.month = rdate.time.month;
678 change.day = rdate.time.day;
679 /* RDATEs with a DATE value inherit the time from
680 the DTSTART. */
681 if (icaltime_is_date(rdate.time)) {
682 change.hour = dtstart.hour;
683 change.minute = dtstart.minute;
684 change.second = dtstart.second;
685 } else {
686 change.hour = rdate.time.hour;
687 change.minute = rdate.time.minute;
688 change.second = rdate.time.second;
689
690 /* The spec was a bit vague about whether RDATEs were in local
691 time or UTC so we support both to be safe. So if it is in
692 UTC we have to add the UTC offset to get a local time. */
693 if (!icaltime_is_utc(rdate.time)) {
694 icaltimezone_adjust_change(&change, 0, 0, 0, -change.prev_utc_offset);
695 }
696 }
697
698#ifdef ICALTIMEZONE_DEBUG_PRINT
699 printf(" Appending RDATE element (Y/M/D): %i/%02i/%02i %i:%02i:%02i\n",
700 change.year, change.month, change.day,
701 change.hour, change.minute, change.second);
702#endif
703
704 icalarray_append(changes, &change);
705 break;
706 case ICAL_RRULE_PROPERTY:
707 rrule = icalproperty_get_rrule(prop);
708 if (rrule) {
709 rrule = icalrecurrencetype_clone(rrule);
710 }
711
712 if (rrule) {
713 /* If the rrule UNTIL value is set and is in UTC, we convert it to
714 a local time, since the recurrence code has no way to convert
715 it itself. */
716 if (!icaltime_is_null_time(rrule->until) && icaltime_is_utc(rrule->until)) {
717#ifdef ICALTIMEZONE_DEBUG_PRINT
718 printf(" Found RRULE UNTIL in UTC.\n");
719#endif
720
721 /* To convert from UTC to a local time, we use the TZOFFSETFROM
722 since that is the offset from UTC that will be in effect
723 when each of the RRULE occurrences happens. */
724 icaltime_adjust(&rrule->until, 0, 0, 0, change.prev_utc_offset);
725 rrule->until.zone = NULL;
726 }
727
728 /* Add the dtstart to changes, otherwise some oddly-defined VTIMEZONE
729 components can cause the first year to get skipped. */
730 change.year = dtstart.year;
731 change.month = dtstart.month;
732 change.day = dtstart.day;
733 change.hour = dtstart.hour;
734 change.minute = dtstart.minute;
735 change.second = dtstart.second;
736
737#ifdef ICALTIMEZONE_DEBUG_PRINT
738 printf(" Appending RRULE element (Y/M/D): %i/%02i/%02i %i:%02i:%02i\n",
739 change.year, change.month, change.day,
740 change.hour, change.minute, change.second);
741#endif
742
743 icaltimezone_adjust_change(&change, 0, 0, 0, -change.prev_utc_offset);
744
745 icalarray_append(changes, &change);
746
747 rrule_iterator = icalrecur_iterator_new(rrule, dtstart);
748 for (size_t rrule_iterator_count = 0; rrule_iterator && rrule_iterator_count < max_rrule_search; rrule_iterator_count++) {
749 occ = icalrecur_iterator_next(rrule_iterator);
750 /* Skip dtstart since we just added it */
751 if (icaltime_compare(dtstart, occ) == 0) {
752 continue;
753 }
754 if (occ.year > end_year || icaltime_is_null_time(occ)) {
755 break;
756 }
757 change.year = occ.year;
758 change.month = occ.month;
759 change.day = occ.day;
760 change.hour = occ.hour;
761 change.minute = occ.minute;
762 change.second = occ.second;
763
764#ifdef ICALTIMEZONE_DEBUG_PRINT
765 printf(" Appending RRULE element (Y/M/D): %i/%02i/%02i %i:%02i:%02i\n",
766 change.year, change.month, change.day,
767 change.hour, change.minute, change.second);
768#endif
769
770 icaltimezone_adjust_change(&change, 0, 0, 0, -change.prev_utc_offset);
771
772 icalarray_append(changes, &change);
773 }
774
775 icalrecur_iterator_free(rrule_iterator);
776 icalrecurrencetype_unref(rrule);
777 }
778 break;
779 default:
780 break;
781 }
782
783 prop = icalcomponent_get_next_property(comp, ICAL_ANY_PROPERTY);
784 }

791{

796 change2 = (const icaltimezonechange *)elem2;

799 retval = -1;
800 } else if (change1->year > change2->year) {
801 retval = 1;
802 } else if (change1->month < change2->month) {
803 retval = -1;
804 } else if (change1->month > change2->month) {
805 retval = 1;
806 } else if (change1->day < change2->day) {
807 retval = -1;
808 } else if (change1->day > change2->day) {

814 } else if (change1->minute < change2->minute) {

816 } else if (change1->minute > change2->minute) {
817 retval = 1;
818 } else if (change1->second < change2->second) {
819 retval = -1;
820 } else if (change1->second > change2->second) {
821 retval = 1;

829void icaltimezone_convert_time(struct icaltimetype *tt,
830 icaltimezone *from_zone, icaltimezone *to_zone)
831{
832 int utc_offset, is_daylight;
833
834 /* If the time is a DATE value or both timezones are the same, or we are
835 converting a floating time, we don't need to do anything. */
836 if (icaltime_is_date(*tt) || from_zone == to_zone || from_zone == NULL) {
837 return;
838 }
839
840 /* Convert the time to UTC by getting the UTC offset and subtracting it. */
841 utc_offset = icaltimezone_get_utc_offset(from_zone, tt, NULL);
842 icaltime_adjust(tt, 0, 0, 0, -utc_offset);
843
844 /* Now we convert the time to the new timezone by getting the UTC offset
845 of our UTC time and adding it. */
846 utc_offset = icaltimezone_get_utc_offset_of_utc_time(to_zone, tt, &is_daylight);
847 tt->is_daylight = is_daylight;
848 tt->zone = to_zone;
849 icaltime_adjust(tt, 0, 0, 0, utc_offset);
850}

852int icaltimezone_get_utc_offset(icaltimezone *zone, const struct icaltimetype *tt, int *is_daylight)
853{
854 const icaltimezonechange *zone_change;
855 const icaltimezonechange *prev_zone_change;
856 icaltimezonechange tt_change = {0}, tmp_change;
857 size_t change_num, change_num_to_use;
858 int found_change;
859 int step, utc_offset_change, cmp;
860 int want_daylight;
861
862 if (tt == NULL || tt->year > ICALTIMEZONE_MAX_YEAR) {
863 return 0;
864 }
865
866 if (is_daylight) {
867 *is_daylight = 0;
868 }
869
870 /* For local times and UTC return 0. */
871 if (zone == NULL || zone == &utc_timezone) {
872 return 0;
873 }
874
875 /* Use the builtin icaltimezone if possible. */
876 if (zone->builtin_timezone) {
877 zone = zone->builtin_timezone;
878 }
879
880 icaltimezone_changes_lock();
881
882 /* Make sure the changes array is expanded up to the given time. */
883 icaltimezone_ensure_coverage(zone, tt->year);
884
885 if (!zone->changes || zone->changes->num_elements == 0) {
886 icaltimezone_changes_unlock();
887 return 0;
888 }
889
890 /* Copy the time parts of the icaltimetype to an icaltimezonechange so we
891 can use our comparison function on it. */
892 tt_change.year = tt->year;
893 tt_change.month = tt->month;
894 tt_change.day = tt->day;
895 tt_change.hour = tt->hour;
896 tt_change.minute = tt->minute;
897 tt_change.second = tt->second;
898
899 /* This should find a change close to the time, either the change before
900 it or the change after it. */
901 change_num = icaltimezone_find_nearby_change(zone, &tt_change);
902
903 /* Now move backwards or forwards to find the timezone change that applies
904 to tt. It should only have to do 1 or 2 steps. */
905 zone_change = icalarray_element_at(zone->changes, change_num);
906 step = 1;
907 found_change = 0;
908 change_num_to_use = (size_t)-1; // invalid on purpose
909 for (;;) {
910 /* Copy the change, so we can adjust it. */
911 tmp_change = *zone_change;
912
913 /* If the clock is going backward, check if it is in the region of time
914 that is used twice. If it is, use the change with the daylight
915 setting which matches tt, or use standard if we don't know. */
916 if (tmp_change.utc_offset < tmp_change.prev_utc_offset) {
917 /* If the time change is at 2:00AM local time and the clock is
918 going back to 1:00AM we adjust the change to 1:00AM. We may
919 have the wrong change but we'll figure that out later. */
920 icaltimezone_adjust_change(&tmp_change, 0, 0, 0, tmp_change.utc_offset);
921 } else {
922 icaltimezone_adjust_change(&tmp_change, 0, 0, 0, tmp_change.prev_utc_offset);
923 }
924
925 cmp = icaltimezone_compare_change_fn(&tt_change, &tmp_change);
926
927 /* If the given time is on or after this change, then this change may
928 apply, but we continue as a later change may be the right one.
929 If the given time is before this change, then if we have already
930 found a change which applies we can use that, else we need to step
931 backwards. */
932 if (cmp >= 0) {
933 found_change = 1;
934 change_num_to_use = change_num;
935 } else {
936 step = -1;
937 }
938
939 /* If we are stepping backwards through the changes and we have found
940 a change that applies, then we know this is the change to use so
941 we exit the loop. */
942 if (step == -1 && found_change == 1) {
943 break;
944 }
945
946 /* If we go past the start of the changes array, then we have no data
947 for this time so we return the prev UTC offset. */
948 if (change_num == 0 && step < 0) {
949 if (is_daylight) {
950 *is_daylight = !tmp_change.is_daylight;
951 }
952
953 icaltimezone_changes_unlock();
954
955 return tmp_change.prev_utc_offset;
956 }
957
958 change_num += (size_t)step;
959
960 if (change_num >= zone->changes->num_elements) {
961 break;
962 }
963
964 zone_change = icalarray_element_at(zone->changes, change_num);
965 }
966
967 /* If we didn't find a change to use, then we have a bug! */
968 icalerror_assert(found_change == 1, "No applicable timezone change found");
969
970 /* Now we just need to check if the time is in the overlapped region of
971 time when clocks go back. */
972 zone_change = icalarray_element_at(zone->changes, change_num_to_use);
973
974 utc_offset_change = zone_change->utc_offset - zone_change->prev_utc_offset;
975 if (utc_offset_change < 0 && change_num_to_use > 0) {
976 tmp_change = *zone_change;
977 icaltimezone_adjust_change(&tmp_change, 0, 0, 0, tmp_change.prev_utc_offset);
978
979 if (icaltimezone_compare_change_fn(&tt_change, &tmp_change) < 0) {
980 /* The time is in the overlapped region, so we may need to use
981 either the current zone_change or the previous one. If the
982 time has the is_daylight field set we use the matching change,
983 else we use the change with standard time. */
984 prev_zone_change = icalarray_element_at(zone->changes, change_num_to_use - 1);
985
986 /* I was going to add an is_daylight flag to struct icaltimetype,
987 but iCalendar doesn't let us distinguish between standard and
988 daylight time anyway, so there's no point. So we just use the
989 standard time instead. */
990 want_daylight = (tt->is_daylight == 1) ? 1 : 0;
991
992#ifdef ICALTIMEZONE_DEBUG_PRINT
993 if (zone_change->is_daylight == prev_zone_change->is_daylight) {
994 printf(" **** Same is_daylight setting\n");
995 }
996#endif
997
998 if (zone_change->is_daylight != want_daylight &&
999 prev_zone_change->is_daylight == want_daylight) {
1000 zone_change = prev_zone_change;
1001 }
1002 }
1003 }
1004
1005 /* Now we know exactly which timezone change applies to the time, so
1006 we can return the UTC offset and whether it is a daylight time. */
1007 if (is_daylight) {
1008 *is_daylight = zone_change->is_daylight;
1009 }
1010 utc_offset_change = zone_change->utc_offset;
1011
1012 icaltimezone_changes_unlock();
1013
1014 return utc_offset_change;
1015}

1135static size_t icaltimezone_find_nearby_change(icaltimezone *zone, const icaltimezonechange *change)
1136{
1137 size_t lower, middle, upper;
1138
1139 /* Do a simple binary search. */
1140 lower = middle = 0;
1141 upper = zone->changes->num_elements;
1142
1143 while (lower < upper) {
1144 middle = (lower + upper) / 2;
1145 const icaltimezonechange *zone_change = icalarray_element_at(zone->changes, middle);
1146 int cmp = icaltimezone_compare_change_fn(change, zone_change);
1147 if (cmp == 0) {
1148 break;
1149 } else if (cmp < 0) {
1150 upper = middle;
1151 } else {
1152 lower = middle + 1;
1153 }
1154 }
1155
1156 return middle;
1157}

1165static void icaltimezone_adjust_change(icaltimezonechange *tt,
1166 int days, int hours, int minutes, int seconds)
1167{
1168 int second, minute, hour, day;
1169 int minutes_overflow, hours_overflow, days_overflow;
1170
1171 /* Add on the seconds. */
1172 second = tt->second + seconds;
1173 tt->second = second % 60;
1174 minutes_overflow = second / 60;
1175 if (tt->second < 0) {
1176 tt->second += 60;
1177 minutes_overflow--;
1178 }
1179
1180 /* Add on the minutes. */
1181 minute = tt->minute + minutes + minutes_overflow;
1182 tt->minute = minute % 60;
1183 hours_overflow = minute / 60;
1184 if (tt->minute < 0) {
1185 tt->minute += 60;
1186 hours_overflow--;
1187 }
1188
1189 /* Add on the hours. */
1190 hour = tt->hour + hours + hours_overflow;
1191 tt->hour = hour % 24;
1192 days_overflow = hour / 24;
1193 if (tt->hour < 0) {
1194 tt->hour += 24;
1195 days_overflow--;
1196 }
1197
1198 /* Add on the days. */
1199 day = tt->day + days + days_overflow;
1200 if (day > 0) {
1201 for (;;) {
1202 int days_in_month = icaltime_days_in_month(tt->month, tt->year);
1203 if (day <= days_in_month) {
1204 break;
1205 }
1206
1207 tt->month++;
1208 if (tt->month >= 13) {
1209 tt->year++;
1210 tt->month = 1;
1211 }
1212
1213 day -= days_in_month;
1214 }
1215 } else {
1216 while (day <= 0) {
1217 if (tt->month == 1) {
1218 tt->year--;
1219 tt->month = 12;
1220 } else {
1221 tt->month--;
1222 }
1223
1224 day += icaltime_days_in_month(tt->month, tt->year);
1225 }
1226 }
1227 tt->day = day;
1228}

1614static bool parse_coord(char *coord, int len, int *degrees, int *minutes, int *seconds)
1615{

1618 } else if (len == 6) {

1620 } else if (len == 7) {
1621 sscanf(coord + 1, "%2d%2d%2d", degrees, minutes, seconds);
1622 } else if (len == 8) {
1623 sscanf(coord + 1, "%3d%2d%2d", degrees, minutes, seconds);
1624 } else {
1625 icalerrprintf("Invalid coordinate: %s\n", coord);
1626 return true;
1627 }

1636static bool fetch_lat_long_from_string(const char *str,
1637 int *latitude_degrees, int *latitude_minutes,
1638 int *latitude_seconds,
1639 int *longitude_degrees, int *longitude_minutes,
1640 int *longitude_seconds,
1641 char *location, size_t len_location)
1642{
1643 size_t len;
1644 const char *loc, *temp;
1645 char *sptr, *lat, *lon;
1646
1647 if (!location || !len_location) {
1648 return false;
1649 }
1650
1651 /* We need to parse the latitude/longitude coordinates and location fields */
1652 const size_t len_str = strlen(str);
1653 size_t i = 0;
1654 sptr = (char *)str;
1655 while ((*sptr != '\t') && (*sptr != '\0')) {
1656 sptr++;
1657 i++;
1658 }
1659 if (i >= len_str) {
1660 return false;
1661 }
1662 temp = ++sptr;
1663 while (*sptr != '\t' && *sptr != '\0') {
1664 sptr++;
1665 }
1666 len = (size_t)(ptrdiff_t)(sptr - temp);
1667 if (len == 0) {
1668 return false;
1669 }
1670 lat = (char *)icalmemory_new_buffer(len + 1);
1671 if (!lat) {
1672 return false;
1673 }
1674 memset(lat, '\0', len + 1);
1675 strncpy(lat, temp, len + 1);
1676 lat[len] = '\0';
1677 while ((*sptr != '\t') && (*sptr != '\0')) {
1678 sptr++;
1679 }
1680 loc = ++sptr;
1681 while (!isspace((int)(*sptr)) && (*sptr != '\0')) {
1682 sptr++;
1683 }
1684 len = (size_t)(ptrdiff_t)(sptr - loc);
1685 if (len >= len_location) {
1686 len = len_location - 1;
1687 }
1688 strncpy(location, loc, len);
1689 location[len] = '\0';
1690
1691 lon = lat + 1;
1692 while (*lon != '\0' && *lon != '+' && *lon != '-') {
1693 lon++;
1694 }
1695
1696 if (parse_coord(lat, (int)(lon - lat),
1697 latitude_degrees,
1698 latitude_minutes,
1699 latitude_seconds) == 1 ||
1700 parse_coord(lon, (int)strlen(lon),
1701 longitude_degrees, longitude_minutes, longitude_seconds) == 1) {
1702 icalmemory_free_buffer(lat);
1703 return true;
1704 }
1705
1706 icalmemory_free_buffer(lat);
1707
1708 return false;
1709}

1853 (double)longitude_seconds / 3600;

1860#endif
1861 }
1862#endif // __clang_analyzer__
1863
1864 builtin_timezones = timezones;
1865
1866 icalmemory_free_buffer(filename);
1867 fclose(fp);

1873static void icaltimezone_load_builtin_timezone(icaltimezone *zone)
1874{
1875 icalcomponent *comp = 0, *subcomp;
1876
1877 /* Prevent blocking on mutex lock caused by recursive calls */
1878 if (zone->component) {
1879 return;
1880 }
1881
1882 icaltimezone_builtin_lock();
1883
1884 /* Try again, maybe it had been set by other thread while waiting for the lock */
1885 if (zone->component) {
1886 icaltimezone_builtin_unlock();
1887 return;
1888 }
1889
1890 /* If the location isn't set, it isn't a builtin timezone. */
1891 if (!zone->location || !zone->location[0]) {
1892 icaltimezone_builtin_unlock();
1893 return;
1894 }
1895
1896 if (use_builtin_tzdata) {
1897 char *filename;
1898 size_t filename_len;
1899 FILE *fp;
1900 icalparser *parser;
1901
1902 filename_len = strlen(get_zone_directory_builtin()) + strlen(zone->location) + 6;
1903
1904 filename = (char *)icalmemory_new_buffer(filename_len);
1905 if (!filename) {
1906 icalerror_set_errno(ICAL_NEWFAILED_ERROR);
1907 goto out;
1908 }
1909
1910 snprintf(filename, filename_len, "%s/%s.ics", get_zone_directory_builtin(), zone->location);
1911
1912 fp = fopen(filename, "r");
1913 icalmemory_free_buffer(filename);
1914 if (!fp) {
1915 icalerror_set_errno(ICAL_FILE_ERROR);
1916 goto out;
1917 }
1918
1919 /* ##### B.# Sun, 11 Nov 2001 04:04:29 +1100
1920 this is where the MALFORMEDDATA error is being set, after the call to 'icalparser_parse'
1921 icalerrprintf("** WARNING ** %s: %d %s\n",
1922 __FILE__, __LINE__, icalerror_strerror(icalerrno));
1923 */
1924
1925 parser = icalparser_new();
1926 icalparser_set_gen_data(parser, fp);
1927 comp = icalparser_parse(parser, icaltimezone_load_get_line_fn);
1928 icalparser_free(parser);
1929 fclose(fp);
1930
1931 /* Find the VTIMEZONE component inside the VCALENDAR. There should be 1. */
1932 subcomp = icalcomponent_get_first_component(comp, ICAL_VTIMEZONE_COMPONENT);
1933
1934 if (subcomp) {
1935 icalproperty *prop;
1936
1937 /* Ensure expected TZID */
1938 prop = icalcomponent_get_first_property(subcomp, ICAL_TZID_PROPERTY);
1939 if (prop) {
1940 char *new_tzid;
1941 size_t new_tzid_len;
1942 const char *tzid_prefix = icaltimezone_tzid_prefix();
1943
1944 new_tzid_len = strlen(tzid_prefix) + strlen(zone->location) + 1;
1945 new_tzid = (char *)icalmemory_new_buffer(sizeof(char) * (new_tzid_len + 1));
1946 if (new_tzid) {
1947 snprintf(new_tzid, new_tzid_len, "%s%s", tzid_prefix, zone->location);
1948 icalproperty_set_tzid(prop, new_tzid);
1949 icalmemory_free_buffer(new_tzid);
1950 } else {
1951 icalerror_set_errno(ICAL_NEWFAILED_ERROR);
1952 }
1953 }
1954
1955 /* Ensure expected Location - it's for cases where one VTIMEZONE is shared
1956 between different locations (like Pacific/Midway is Pacific/Pago_Pago).
1957 This updates the properties, thus when the component is converted to
1958 the string and back to the component the Location will still match. */
1959 prop = icalcomponent_get_first_property(subcomp, ICAL_LOCATION_PROPERTY);
1960 if (prop) {
1961 icalproperty_set_location(prop, zone->location);
1962 }
1963
1964 for (prop = icalcomponent_get_first_property(subcomp, ICAL_X_PROPERTY);
1965 prop;
1966 prop = icalcomponent_get_next_property(subcomp, ICAL_X_PROPERTY)) {
1967 const char *name;
1968
1969 name = icalproperty_get_x_name(prop);
1970 if (name && !strcasecmp(name, "X-LIC-LOCATION")) {
1971 icalproperty_set_x(prop, zone->location);
1972 break;
1973 }
1974 }
1975 }
1976 } else {
1977 subcomp = icaltimezone_fetch_timezone(zone->location);
1978 }
1979
1980 if (!subcomp) {
1981 icalerror_set_errno(ICAL_PARSE_ERROR);
1982 goto out;
1983 }
1984
1985 icaltimezone_get_vtimezone_properties(zone, subcomp);
1986
1987 if (use_builtin_tzdata) {
1988 icalcomponent_remove_component(comp, subcomp);
1989 icalcomponent_free(comp);
1990 }
1991
1992out:
1993 icaltimezone_builtin_unlock();
1994}
