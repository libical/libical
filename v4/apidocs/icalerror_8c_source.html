44icalerrorenum *icalerror_icalerrno(void)
45{
46 icalerrorenum *_errno;
47
48 pthread_once(&icalerrno_key_once, icalerrno_key_alloc);
49
50 _errno = (icalerrorenum *)pthread_getspecific(icalerrno_key);
51
52 if (!_errno) {
53 _errno = icalmemory_new_buffer(sizeof(icalerrorenum));
54 *_errno = ICAL_NO_ERROR;
55 pthread_setspecific(icalerrno_key, _errno);
56 }
57 return _errno;
58}

115void icalerror_set_errno(icalerrorenum x)
116{
117 icalerrno = x;

119 (icalerror_get_error_state(x) == ICAL_ERROR_DEFAULT && icalerror_errors_are_fatal == 1)) {

174icalerrorenum icalerror_error_from_string(const char *str)
175{
176 int i;
177
178 for (i = 0; string_map[i].error != ICAL_UNKNOWN_ERROR; i++) {
179 if (strcmp(string_map[i].str, str) == 0) {
180 break;
181 }
182 }
183
184 return string_map[i].error;
185}

187icalerrorstate icalerror_supress(const char *error)
188{
189 icalerrorenum e = icalerror_error_from_string(error);
190 icalerrorstate es;
191
192 if (e == ICAL_NO_ERROR) {
193 return ICAL_ERROR_UNKNOWN;
194 }
195
196 es = icalerror_get_error_state(e);
197 icalerror_set_error_state(e, ICAL_ERROR_NONFATAL);

253void icalerror_backtrace(void)
254{
255#if defined(HAVE_BACKTRACE)
256 void *stack_frames[50];
257 int i, num;
258 size_t size;
259 char **strings;
260
261 size = sizeof(stack_frames) / sizeof(stack_frames[0]);
262 num = backtrace(stack_frames, (int)size);
263 strings = backtrace_symbols(stack_frames, num);
264 for (i = 0; i < num; i++) {
265 if (strings != NULL) {
266 icalerrprintf("%s\n", strings[i]);
267 } else {
268 icalerrprintf("%p\n", stack_frames[i]);
269 }
270 }
271 free((void *)strings); /* Not icalmemory_free_buffer(), allocated by backtrace_symbols() */
272#endif
273}
