55static vcardcomponent *vcardcomponent_new_impl(vcardcomponent_kind kind)
56{
57 vcardcomponent *comp;
58
59 if (!vcardcomponent_kind_is_valid(kind)) {
60 return NULL;
61 }
62
63 if ((comp = (vcardcomponent *)icalmemory_new_buffer(sizeof(vcardcomponent))) == 0) {
64 icalerror_set_errno(ICAL_NEWFAILED_ERROR);
65 return 0;
66 }
67
68 memset(comp, 0, sizeof(vcardcomponent));
69
70 strcpy(comp->id, "comp");
71
72 comp->kind = kind;
73 comp->properties = icalpvl_newlist();
74 comp->components = icalpvl_newlist();
75
76 return comp;
77}

114{
115 vcardcomponent *clone;
116 const vcardproperty *p;
117 icalpvl_elem itr;
118
119 icalerror_check_arg_rz((old != 0), "component");
120

122

129 vcardcomponent_add_property(clone, vcardproperty_clone(p));
130 }

135void vcardcomponent_free(vcardcomponent *c)
136{
137 vcardcomponent *comp;
138
139 icalerror_check_arg_rv((c != 0), "component");
140
141 if (c->parent != 0) {
142 return;
143 }
144
145 if (c->properties != 0) {
146 vcardproperty *prop;
147 while ((prop = icalpvl_pop(c->properties)) != 0) {
148 vcardproperty_set_parent(prop, 0);
149 vcardproperty_free(prop);
150 }
151 icalpvl_free(c->properties);
152 }
153
154 while ((comp = icalpvl_data(icalpvl_head(c->components))) != 0) {
155 vcardcomponent_remove_component(c, comp);
156 vcardcomponent_free(comp);
157 }
158
159 icalpvl_free(c->components);
160
161 if (c->x_name != 0) {
162 icalmemory_free_buffer(c->x_name);
163 }
164
165 c->kind = VCARD_NO_COMPONENT;
166 c->properties = 0;
167 c->property_iterator = 0;
168 c->components = 0;
169 c->component_iterator = 0;
170 c->x_name = 0;
171 c->id[0] = 'X';
172
173 icalmemory_free_buffer(c);
174}

187char *vcardcomponent_as_vcard_string_r(vcardcomponent *comp)
188{
189 char *buf;
190 char *tmp_buf;
191 size_t buf_size = 1024;
192 char *buf_ptr = 0;
193 icalpvl_elem itr;
194
195 /* RFC6350 explicitly says that the newline is *ALWAYS* a \r\n (CRLF)!!!! */
196 const char newline[] = "\r\n";
197
198 vcardcomponent *c;
199 vcardproperty *p;
200 vcardcomponent_kind kind = vcardcomponent_isa(comp);
201
202 const char *kind_string = NULL;
203
204 icalerror_check_arg_rz((comp != 0), "component");
205 icalerror_check_arg_rz((kind != VCARD_NO_COMPONENT),
206 "component kind is VCARD_NO_COMPONENT");
207
208 if (kind != VCARD_X_COMPONENT) {
209 kind_string = vcardcomponent_kind_to_string(kind);
210 } else {
211 kind_string = comp->x_name;
212 }
213
214 icalerror_check_arg_rz((kind_string != 0), "Unknown kind of component");
215
216 buf = icalmemory_new_buffer(buf_size);
217 if (buf == NULL) {
218 return NULL;
219 }
220
221 buf_ptr = buf;
222
223 if (kind != VCARD_XROOT_COMPONENT) {
224 icalmemory_append_string(&buf, &buf_ptr, &buf_size, "BEGIN:");
225 icalmemory_append_string(&buf, &buf_ptr, &buf_size, kind_string);
226 icalmemory_append_string(&buf, &buf_ptr, &buf_size, newline);
227
228 for (itr = icalpvl_head(comp->properties); itr != 0; itr = icalpvl_next(itr)) {
229 p = (vcardproperty *)icalpvl_data(itr);
230
231 icalerror_assert((p != 0), "Got a null property");
232 tmp_buf = vcardproperty_as_vcard_string_r(p);
233
234 icalmemory_append_string(&buf, &buf_ptr, &buf_size, tmp_buf);
235 icalmemory_free_buffer(tmp_buf);
236 }
237 }
238
239 for (itr = icalpvl_head(comp->components); itr != 0; itr = icalpvl_next(itr)) {
240 c = (vcardcomponent *)icalpvl_data(itr);
241
242 tmp_buf = vcardcomponent_as_vcard_string_r(c);
243 if (tmp_buf != NULL) {
244 icalmemory_append_string(&buf, &buf_ptr, &buf_size, tmp_buf);
245 icalmemory_free_buffer(tmp_buf);
246 }
247 }
248
249 if (kind != VCARD_XROOT_COMPONENT) {
250 icalmemory_append_string(&buf, &buf_ptr, &buf_size, "END:");
251 icalmemory_append_string(&buf, &buf_ptr, &buf_size, kind_string);
252 icalmemory_append_string(&buf, &buf_ptr, &buf_size, newline);
253 }
254
255 return buf;
256}

284void vcardcomponent_add_property(vcardcomponent *comp, vcardproperty *property)
285{
286 icalerror_check_arg_rv((comp != 0), "component");
287 icalerror_check_arg_rv((property != 0), "property");
288
289 icalerror_assert((!vcardproperty_get_parent(property)),
290 "The property has already been added to a vcard. "
291 "Remove the property with vcardcomponent_remove_property "
292 "before calling vcardcomponent_add_property");
293
294 vcardproperty_set_parent(property, comp);
295
296 icalpvl_push(comp->properties, property);
297
298 if (vcardproperty_isa(property) == VCARD_VERSION_PROPERTY) {
299 comp->versionp = property;
300 }

303void vcardcomponent_remove_property(vcardcomponent *comp, vcardproperty *property)
304{
305 icalpvl_elem itr, next_itr;
306
307 icalerror_check_arg_rv((comp != 0), "component");
308 icalerror_check_arg_rv((property != 0), "property");
309
310 if (vcardproperty_get_parent(property) == 0) {
311 return;
312 }
313
314 if (vcardproperty_isa(property) == VCARD_VERSION_PROPERTY) {
315 comp->versionp = 0;
316 }
317
318 for (itr = icalpvl_head(comp->properties); itr != 0; itr = next_itr) {
319 next_itr = icalpvl_next(itr);
320
321 if (icalpvl_data(itr) == (void *)property) {
322 if (comp->property_iterator == itr) {
323 comp->property_iterator = icalpvl_next(itr);
324 }
325
326 (void)icalpvl_remove(comp->properties, itr);
327 vcardproperty_set_parent(property, 0);
328 }
329 }

332int vcardcomponent_count_properties(vcardcomponent *comp,
333 vcardproperty_kind kind,
334 int ignore_alts)
335{
336 int count = 0;
337 icalpvl_elem itr;
338 vcardstrarray *altids = NULL;
339
340 icalerror_check_arg_rz((comp != 0), "component");
341
342 if (ignore_alts) {
343 altids = vcardstrarray_new(2);
344 }
345
346 for (itr = icalpvl_head(comp->properties); itr != 0; itr = icalpvl_next(itr)) {
347 vcardproperty *prop = (vcardproperty *)icalpvl_data(itr);
348
349 if (kind == VCARD_ANY_PROPERTY || kind == vcardproperty_isa(prop)) {
350 if (ignore_alts) {
351 /* Like-properties having the same ALTID only get counted once */
352 vcardparameter *param =
353 vcardproperty_get_first_parameter(prop,
354 VCARD_ALTID_PARAMETER);
355 if (param) {
356 const char *altid = vcardparameter_get_altid(param);
357
358 if (vcardstrarray_find(altids, altid) >= vcardstrarray_size(altids)) {
359 continue;
360 }
361
362 vcardstrarray_append(altids, altid);
363 }
364 }
365 count++;
366 }
367 }
368
369 if (ignore_alts) {
370 vcardstrarray_free(altids);
371 }
372
373 return count;
374}

443void vcardcomponent_remove_component(vcardcomponent *parent,
444 vcardcomponent *child)
445{
446 icalpvl_elem itr, next_itr;
447
448 icalerror_check_arg_rv((parent != 0), "parent");
449 icalerror_check_arg_rv((child != 0), "child");
450
451 for (itr = icalpvl_head(parent->components); itr != 0; itr = next_itr) {
452 next_itr = icalpvl_next(itr);
453
454 if (icalpvl_data(itr) == (void *)child) {
455 if (parent->component_iterator == itr) {
456 /* Don't let the current iterator become invalid */
457
458 /* HACK. The semantics for this are troubling. */
459 parent->component_iterator = icalpvl_next(parent->component_iterator);
460 }
461 (void)icalpvl_remove(parent->components, itr);
462 child->parent = 0;
463 break;
464 }
465 }
466}

639static int prop_compare(void *a, void *b)
640{
641 const vcardproperty *p1 = (vcardproperty *)a;
642 const vcardproperty *p2 = (vcardproperty *)b;
643 vcardproperty_kind k1 = vcardproperty_isa(p1);
644 vcardproperty_kind k2 = vcardproperty_isa(p2);
645 int r = (int)(k1 - k2);
646
647 if (r == 0) {
648 if (k1 == VCARD_X_PROPERTY) {
649 r = strcmp(vcardproperty_get_x_name(p1),
650 vcardproperty_get_x_name(p2));
651 }
652
653 if (r == 0) {
654 r = strcmpsafe(vcardproperty_get_value_as_string(p1),
655 vcardproperty_get_value_as_string(p2));
656 }
657 }
658 /* Always sort VERSION first */
659 else if (k1 == VCARD_VERSION_PROPERTY) {
660 r = -1;
661 } else if (k2 == VCARD_VERSION_PROPERTY) {
662 r = 1;
663 }
664
665 return r;
666}
667
668static int prop_kind_compare(vcardproperty_kind kind,
669 vcardcomponent *c1, vcardcomponent *c2)
670{
671 const vcardproperty *p1 = vcardcomponent_get_first_property(c1, kind);
672 const vcardproperty *p2 = vcardcomponent_get_first_property(c2, kind);
673
674 if (p1 && p2) {
675 return strcmpsafe(vcardproperty_get_value_as_string(p1),
676 vcardproperty_get_value_as_string(p2));
677 } else if (p1) {
678 return -1;
679 } else if (p2) {
680 return 1;
681 }
682
683 return 0;
684}

686static int comp_compare(void *a, void *b)
687{
688 vcardcomponent *c1 = (vcardcomponent *)a;
689 vcardcomponent *c2 = (vcardcomponent *)b;
690 vcardcomponent_kind k1 = vcardcomponent_isa(c1);
691 vcardcomponent_kind k2 = vcardcomponent_isa(c2);
692 int r = (int)(k1 - k2);
693
694 if (r == 0) {
695 if (k1 == VCARD_VCARD_COMPONENT) {
696 vcardproperty_kind prop_kinds[] = {
697 VCARD_VERSION_PROPERTY,
698 VCARD_N_PROPERTY,
699 VCARD_FN_PROPERTY,
700 VCARD_NICKNAME_PROPERTY,
701 VCARD_ORG_PROPERTY,
702 /* XXX What else should we compare? */
703 VCARD_NO_PROPERTY};
704
705 for (int i = 0; r == 0 && prop_kinds[i] != VCARD_NO_PROPERTY; i++) {
706 r = prop_kind_compare(prop_kinds[i], c1, c2);
707 }
708 } else {
709 r = strcmp(c1->x_name, c2->x_name);
710 }
711
712 if (r == 0) {
713 r = prop_kind_compare(VCARD_UID_PROPERTY, c1, c2);
714
715 if (r == 0) {
716 /* XXX Anything else to compare? */
717 }
718 }
719 }
720
721 return r;
722}

724void vcardcomponent_normalize(vcardcomponent *comp)
725{
726 icalpvl_list sorted_props = icalpvl_newlist();
727 icalpvl_list sorted_comps = icalpvl_newlist();
728 vcardproperty *prop;
729 vcardcomponent *sub;
730
731 /* Normalize properties into sorted list */
732 while ((prop = icalpvl_pop(comp->properties)) != 0) {
733 int nparams, remove = 0;
734
735 vcardproperty_normalize(prop);
736
737 nparams = vcardproperty_count_parameters(prop);
738
739 /* Remove unparameterized properties having default values */
740 if (nparams == 0) {
741 switch (vcardproperty_isa(prop)) {
742 case VCARD_KIND_PROPERTY:
743 if (vcardproperty_get_kind(prop) == VCARD_KIND_INDIVIDUAL) {
744 remove = 1;
745 }
746 break;
747
748 default:
749 break;
750 }
751 }
752
753 if (remove) {
754 vcardproperty_set_parent(prop, 0); // MUST NOT have a parent to free
755 vcardproperty_free(prop);
756 } else {
757 icalpvl_insert_ordered(sorted_props, prop_compare, prop);
758 }
759 }
760
761 icalpvl_free(comp->properties);
762 comp->properties = sorted_props;
763
764 /* Normalize sub-components into sorted list */
765 while ((sub = icalpvl_pop(comp->components)) != 0) {
766 vcardcomponent_normalize(sub);
767 icalpvl_insert_ordered(sorted_comps, comp_compare, sub);
768 }
769
770 icalpvl_free(comp->components);
771 comp->components = sorted_comps;
772}

777static void comp_to_v4(vcardcomponent *impl)
778{
779 icalpvl_elem itr, next;
780
781 for (itr = icalpvl_head(impl->properties); itr != 0; itr = next) {
782 vcardproperty *prop = (vcardproperty *)icalpvl_data(itr);
783 vcardproperty_kind pkind = vcardproperty_isa(prop);
784 vcardvalue *value = vcardproperty_get_value(prop);
785 vcardvalue_kind vkind = vcardvalue_isa(value);
786 vcardparameter *param;
787 vcardenumarray *types = NULL;
788
789 next = icalpvl_next(itr);
790
791 /* Remove TYPE=PREF and replace with PREF=1 (if no existing (PREF=) */
792 for (param = vcardproperty_get_first_parameter(prop,
793 VCARD_TYPE_PARAMETER);
794 param;
795 param = vcardproperty_get_next_parameter(prop,
796 VCARD_TYPE_PARAMETER)) {
797 vcardenumarray_element pref = {VCARD_TYPE_PREF, NULL};
798 size_t i;
799
800 types = vcardparameter_get_type(param);
801 i = vcardenumarray_find(types, &pref);
802 if (i < vcardenumarray_size(types)) {
803 vcardenumarray_remove_element_at(types, i);
804 if (!vcardenumarray_size(types)) {
805 vcardproperty_remove_parameter_by_ref(prop, param);
806 }
807
808 param = vcardproperty_get_first_parameter(prop,
809 VCARD_PREF_PARAMETER);
810 if (!param) {
811 param = vcardparameter_new_pref(1);
812 vcardproperty_add_parameter(prop, param);
813 }
814
815 break;
816 }
817 }
818
819 switch (pkind) {
820 case VCARD_VERSION_PROPERTY:
821 vcardproperty_set_version(prop, VCARD_VERSION_40);
822 break;
823
824 case VCARD_BDAY_PROPERTY:
825 case VCARD_DEATHDATE_PROPERTY:
826 case VCARD_ANNIVERSARY_PROPERTY:
827 for (param = vcardproperty_get_first_parameter(prop,
828 VCARD_X_PARAMETER);
829 param;
830 param = vcardproperty_get_next_parameter(prop,
831 VCARD_X_PARAMETER)) {
832 const char *name = vcardparameter_get_xname(param);
833
834 /* This appears in the wild for v3 date with missing year */
835 if (name && !strcasecmp(name, "X-APPLE-OMIT-YEAR")) {
836 vcardtimetype dt = vcardproperty_get_bday(prop);
837
838 dt.year = -1;
839 vcardproperty_set_bday(prop, dt);
840 vcardproperty_remove_parameter_by_ref(prop, param);
841 break;
842 }
843 }
844 break;
845
846 case VCARD_GEO_PROPERTY:
847 if (vkind != VCARD_X_VALUE) {
848 vcardgeotype geo = vcardvalue_get_geo(value);
849
850 if (!geo.uri) {
851 /* Convert STRUCTURED value kind to geo: URI */
852 int n = snprintf(NULL, 0, "geo:%s,%s",
853 geo.coords.lat, geo.coords.lon);
854
855 size_t size = (size_t)n + 1;
856 char *buf = icalmemory_new_buffer(size);
857 snprintf(buf, size, "geo:%s,%s",
858 geo.coords.lat, geo.coords.lon);
859
860 geo.uri = buf;
861 geo.coords.lat[0] = '\0';
862 geo.coords.lon[0] = '\0';
863 vcardvalue_set_geo(value, geo);
864 icalmemory_free_buffer(buf);
865 }
866 }
867 break;
868
869 case VCARD_KEY_PROPERTY:
870 case VCARD_LOGO_PROPERTY:
871 case VCARD_PHOTO_PROPERTY:
872 case VCARD_SOUND_PROPERTY: {
873 char *mediatype = NULL;
874 param = vcardproperty_get_first_parameter(prop,
875 VCARD_TYPE_PARAMETER);
876 if (param) {
877 types = vcardparameter_get_type(param);
878 for (size_t i = 0; i < vcardenumarray_size(types); i++) {
879 const vcardenumarray_element *type =
880 vcardenumarray_element_at(types, i);
881
882 if (type->xvalue) {
883 /* Copy and lowercase the mediatype */
884 char *c;
885
886 switch (pkind) {
887 case VCARD_LOGO_PROPERTY:
888 case VCARD_PHOTO_PROPERTY:
889 mediatype = icalmemory_strdup("image/");
890 break;
891 case VCARD_SOUND_PROPERTY:
892 mediatype = icalmemory_strdup("audio/");
893 break;
894 default:
895 mediatype = icalmemory_strdup("application/");
896 break;
897 }
898
899 size_t size = strlen(mediatype);
900 char *buf_ptr = mediatype + size;
901 icalmemory_append_string(&mediatype, &buf_ptr,
902 &size, type->xvalue);
903 for (c = mediatype; *c; c++) {
904 *c = (char)tolower((int)*c);
905 }
906
907 /* Remove this TYPE */
908 vcardenumarray_remove_element_at(types, i);
909 if (!vcardenumarray_size(types)) {
910 vcardproperty_remove_parameter_by_ref(prop, param);
911 }
912 break;
913 }
914 }
915 }
916
917 const char *data = vcardvalue_get_uri(value);
918 if (data && !strchr(data, ':')) {
919 /* Convert base64-encoded TEXT value kind to data: URI */
920 size_t size = strlen(data) + 7; // "data:," + NUL
921
922 param = vcardproperty_get_first_parameter(prop,
923 VCARD_ENCODING_PARAMETER);
924 if (param) {
925 size += 7; // ";base64
926 }
927
928 char *buf = icalmemory_new_buffer(size);
929 char *buf_ptr = buf;
930
931 icalmemory_append_string(&buf, &buf_ptr, &size, "data:");
932 if (mediatype) {
933 icalmemory_append_string(&buf, &buf_ptr, &size, mediatype);
934 icalmemory_free_buffer(mediatype);
935 }
936 if (param) {
937 icalmemory_append_string(&buf, &buf_ptr, &size, ";base64");
938 vcardproperty_remove_parameter_by_ref(prop, param);
939 }
940 icalmemory_append_char(&buf, &buf_ptr, &size, ',');
941 icalmemory_append_string(&buf, &buf_ptr, &size, data);
942
943 value->kind = VCARD_URI_VALUE;
944 vcardvalue_set_uri(value, buf);
945
946 icalmemory_free_buffer(buf);
947 } else if (mediatype) {
948 param = vcardparameter_new_mediatype(mediatype);
949 vcardproperty_add_parameter(prop, param);
950
951 icalmemory_free_buffer(mediatype);
952 }
953 break;
954 }
955
956 case VCARD_UID_PROPERTY: {
957 /* Does it look like a URI (the default)? */
958 const char *data = vcardvalue_get_text(value);
959 if (!strncasecmp(data, "urn:uuid:", 9) ||
960 !strncasecmp(data, "mailto:", 7) ||
961 !strncasecmp(data, "http://", 7) ||
962 !strncasecmp(data, "https://", 8) ||
963 !strncasecmp(data, "ldap://", 7) ||
964 !strncasecmp(data, "ldaps://", 8)) {
965 value->kind = VCARD_URI_VALUE;
966 vcardproperty_remove_parameter_by_kind(prop,
967 VCARD_VALUE_PARAMETER);
968 } else {
969 /* Otherwise, treat it as TEXT */
970 value->kind = VCARD_TEXT_VALUE;
971 }
972 break;
973 }
974
975 case VCARD_X_PROPERTY: {
976 /* Rename X-ADDRESSBOOKSERVER-KIND, X-ADDRESSBOOKSERVER-MEMBER */
977 const char *xname = vcardproperty_get_x_name(prop);
978
979 if (!strncasecmp(xname, "X-ADDRESSBOOKSERVER-", 20)) {
980 vcardproperty_kind kind = vcardproperty_string_to_kind(xname + 20);
981 const char *valstr = vcardvalue_as_vcard_string(value);
982 vcardproperty *new;
983 char *buf = NULL, *buf_ptr;
984 size_t buf_size;
985
986 switch (kind) {
987 case VCARD_MEMBER_PROPERTY:
988 /* strip urn:uuid: prefix and RFC6868-decode string */
989 if (!strncmp(valstr, UUID_PREFIX, UUID_PREFIX_LEN)) {
990 valstr += UUID_PREFIX_LEN;
991 }
992
993 buf_size = strlen(valstr) + 1;
994 buf_ptr = buf = icalmemory_new_buffer(buf_size);
995 icalmemory_append_decoded_string(&buf, &buf_ptr, &buf_size,
996 valstr);
997 valstr = buf;
998
999 _fallthrough();
1000
1001 case VCARD_KIND_PROPERTY:
1002 new = vcardproperty_new(kind);
1003 vcardproperty_set_value_from_string(new, valstr, "NO");
1004 vcardcomponent_add_property(impl, new);
1005 vcardcomponent_remove_property(impl, prop);
1006 vcardproperty_free(prop);
1007 break;
1008
1009 default:
1010 break;
1011 }
1012
1013 icalmemory_free_buffer(buf);
1014 }
1015 break;
1016 }
1017
1018 default:
1019 break;
1020 }
1021 }

1025 vcardproperty *prop;
1026 int level;

1029static void comp_to_v3(vcardcomponent *impl)
1030{
1031 struct pref_prop *pref_props[VCARD_NO_PROPERTY] = {0};
1032 vcardenumarray_element type;
1033 vcardproperty_kind pkind;
1034 icalpvl_elem itr, next;
1035
1036 for (itr = icalpvl_head(impl->properties); itr != 0; itr = next) {
1037 vcardproperty *prop = (vcardproperty *)icalpvl_data(itr);
1038 vcardparameter *val_param =
1039 vcardproperty_get_first_parameter(prop, VCARD_VALUE_PARAMETER);
1040 vcardvalue *value = vcardproperty_get_value(prop);
1041 vcardvalue_kind vkind = vcardvalue_isa(value);
1042 vcardparameter *param;
1043 char *subtype = NULL;
1044 const char *mediatype, *uri, *xname = NULL;
1045
1046 next = icalpvl_next(itr);
1047
1048 /* Find prop with lowest PREF= for each set of like properties */
1049 pkind = vcardproperty_isa(prop);
1050 param = vcardproperty_get_first_parameter(prop, VCARD_PREF_PARAMETER);
1051 if (param && pkind != VCARD_X_PROPERTY) {
1052 int level = vcardparameter_get_pref(param);
1053 struct pref_prop *pp = pref_props[pkind];
1054
1055 if (!pp) {
1056 pp = icalmemory_new_buffer(sizeof(struct pref_prop));
1057 pp->prop = prop;
1058 pp->level = level;
1059 pref_props[pkind] = pp;
1060 } else if (level < pp->level) {
1061 pp->prop = prop;
1062 pp->level = level;
1063 }
1064 }
1065
1066 /* Replace MEDIATYPE with TYPE=<subtype> */
1067 param = vcardproperty_get_first_parameter(prop,
1068 VCARD_MEDIATYPE_PARAMETER);
1069 if (param) {
1070 mediatype = vcardparameter_get_mediatype(param);
1071 subtype = strchr(mediatype, '/');
1072 if (subtype) {
1073 /* Copy and uppercase the subtype */
1074 char *c;
1075
1076 subtype = icalmemory_strdup(subtype + 1);
1077 for (c = subtype; *c; c++) {
1078 *c = (char)toupper((int)*c);
1079 }
1080
1081 /* Add TYPE parameter */
1082 type.val = VCARD_TYPE_NONE;
1083 type.xvalue = subtype;
1084 vcardproperty_add_type_parameter(prop, &type);
1085
1086 icalmemory_free_buffer(subtype);
1087 }
1088
1089 vcardproperty_remove_parameter_by_ref(prop, param);
1090 }
1091
1092 switch (pkind) {
1093 case VCARD_VERSION_PROPERTY:
1094 vcardproperty_set_version(prop, VCARD_VERSION_30);
1095 break;
1096
1097 case VCARD_BDAY_PROPERTY:
1098 case VCARD_DEATHDATE_PROPERTY:
1099 case VCARD_ANNIVERSARY_PROPERTY: {
1100 vcardtimetype dt = vcardproperty_get_bday(prop);
1101
1102 if (dt.year == -1) {
1103 /* This appears in the wild for v3 date with missing year */
1104 dt.year = 1604;
1105 vcardproperty_set_parameter_from_string(prop,
1106 "X-APPLE-OMIT-YEAR",
1107 "1604");
1108 }
1109 if (dt.hour != -1) {
1110 if (dt.second == -1) {
1111 dt.second = 0;
1112 if (dt.minute == -1) {
1113 dt.minute = 0;
1114 }
1115 }
1116 }
1117
1118 vcardproperty_set_bday(prop, dt);
1119 break;
1120 }
1121
1122 case VCARD_GEO_PROPERTY:
1123 if (vkind != VCARD_X_VALUE) {
1124 vcardgeotype geo = vcardvalue_get_geo(value);
1125
1126 if (geo.uri && !strncmp(geo.uri, "geo:", 4)) {
1127 /* Convert geo: URI to STRUCTURED value kind */
1128 char *buf = icalmemory_strdup(geo.uri);
1129 geo.uri = NULL;
1130 const char *lat = buf + 4;
1131 char *lon = strchr(buf + 4, ',');
1132 if (lon) {
1133 *lon++ = '\0';
1134 }
1135 if (lat && lon) {
1136 strncpy(geo.coords.lat, lat, VCARD_GEO_LEN - 1);
1137 geo.coords.lat[VCARD_GEO_LEN - 1] = '\0';
1138 strncpy(geo.coords.lon, lon, VCARD_GEO_LEN - 1);
1139 geo.coords.lon[VCARD_GEO_LEN - 1] = '\0';
1140 } else {
1141 geo.coords.lat[0] = '\0';
1142 geo.coords.lon[0] = '\0';
1143 }
1144
1145 vcardvalue_set_geo(value, geo);
1146 icalmemory_free_buffer(buf);
1147 }
1148 }
1149 break;
1150
1151 case VCARD_KEY_PROPERTY:
1152 case VCARD_LOGO_PROPERTY:
1153 case VCARD_PHOTO_PROPERTY:
1154 case VCARD_SOUND_PROPERTY:
1155 uri = vcardvalue_get_uri(value);
1156 if (uri && !strncmp("data:", uri, 5)) {
1157 /* Convert data: URI to base64-encoded TEXT value kind */
1158 char *base64, *data = NULL;
1159
1160 char *buf = icalmemory_strdup(uri);
1161 mediatype = buf + 5;
1162 base64 = strstr(mediatype, ";base64,");
1163
1164 if (base64) {
1165 param = vcardparameter_new_encoding(VCARD_ENCODING_B);
1166 vcardproperty_add_parameter(prop, param);
1167
1168 *base64 = '\0';
1169 data = base64 + 8;
1170 } else {
1171 data = strchr(mediatype, ',');
1172 if (data) {
1173 *data++ = '\0';
1174 }
1175 }
1176
1177 subtype = strchr(mediatype, '/');
1178 if (subtype) {
1179 /* Copy and uppercase the subtype */
1180 char *c;
1181
1182 for (c = ++subtype; *c; c++) {
1183 *c = (char)toupper((int)*c);
1184 }
1185
1186 /* Add TYPE parameter */
1187 type.val = VCARD_VERSION_NONE;
1188 type.xvalue = subtype;
1189 vcardproperty_add_type_parameter(prop, &type);
1190 }
1191
1192 value->kind = VCARD_TEXT_VALUE;
1193 vcardvalue_set_text(value, data ? data : "");
1194 icalmemory_free_buffer(buf);
1195 }
1196 break;
1197
1198 case VCARD_KIND_PROPERTY:
1199 /* Rename KIND, MEMBER */
1200 xname = "X-ADDRESSBOOKSERVER-KIND";
1201
1202 _fallthrough();
1203
1204 case VCARD_MEMBER_PROPERTY: {
1205 char *buf = NULL;
1206
1207 const char *xval = vcardvalue_as_vcard_string(value);
1208 if (!xname) {
1209 size_t buf_size = strlen(UUID_PREFIX) + strlen(xval) + 1;
1210 char *buf_ptr = buf = icalmemory_new_buffer(buf_size);
1211
1212 /* RFC6868-encode the uid and prefix with urn:uuid: */
1213 icalmemory_append_string(&buf, &buf_ptr, &buf_size, UUID_PREFIX);
1214 icalmemory_append_encoded_string(&buf, &buf_ptr, &buf_size, xval);
1215 xname = "X-ADDRESSBOOKSERVER-MEMBER";
1216 xval = buf;
1217 }
1218
1219 vcardproperty *xprop = vcardproperty_new_x(xval);
1220 vcardproperty_set_x_name(xprop, xname);
1221 vcardcomponent_add_property(impl, xprop);
1222 vcardcomponent_remove_property(impl, prop);
1223 vcardproperty_free(prop);
1224 icalmemory_free_buffer(buf);
1225 break;
1226 }
1227
1228 case VCARD_TEL_PROPERTY:
1229 uri = vcardvalue_get_uri(value);
1230 if (uri && !strncmp(uri, "tel:", 4)) {
1231 /* Convert tel: URI to TEXT value kind */
1232 char *buf = icalmemory_strdup(uri + 4);
1233
1234 value->kind = VCARD_TEXT_VALUE;
1235 vcardvalue_set_text(value, buf);
1236
1237 if (val_param) {
1238 vcardproperty_remove_parameter_by_ref(prop, val_param);
1239 }
1240
1241 icalmemory_free_buffer(buf);
1242 }
1243 break;
1244
1245 case VCARD_UID_PROPERTY:
1246 /* Treat all values as TEXT (the default) */
1247 value->kind = VCARD_TEXT_VALUE;
1248
1249 if (val_param) {
1250 vcardproperty_remove_parameter_by_ref(prop, val_param);
1251 }
1252 break;
1253
1254 default:
1255 break;
1256 }
1257 }
1258
1259 /* Add TYPE=PREF for each most preferred property */
1260 for (pkind = 0; pkind < VCARD_NO_PROPERTY; ++pkind) {
1261 struct pref_prop *pp = pref_props[pkind];
1262
1263 if (pp) {
1264 type.val = VCARD_TYPE_PREF;
1265 type.xvalue = NULL;
1266
1267 vcardproperty_add_type_parameter(pp->prop, &type);
1268 icalmemory_free_buffer(pp);
1269 }
1270 }
1271}
