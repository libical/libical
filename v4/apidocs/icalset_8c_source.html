136static bool load(const char *file)
137{
138 void *modh;
139 fptr inith;
140 const char *dlerr;
141 icalset *icalset_init_ptr;
142
143 if ((modh = dlopen(file, RTLD_NOW)) == 0) {
144 perror("dlopen");
145 return false;
146 }
147
148 (void)dlerror(); /* clear */
149 inith = (fptr)dlsym(modh, "InitModule");
150 dlerr = dlerror();
151 if (dlerr != NULL) {
152 fprintf(stderr, "dlsym error: %s\n", dlerr);
153 dlclose(modh);
154 return false;
155 }
156
157 while ((icalset_init_ptr = ((inith)())) != 0) {
158 icalpvl_push(icalset_kinds, &icalset_init_ptr);
159 }
160
161 (void)dlerror(); /* clear */
162 dlclose(modh);
163 return true;
164}

166static void icalset_init(void)
167{
168 assert(icalset_kinds == 0);
169 icalset_kinds = icalpvl_newlist();
170
171 icalpvl_push(icalset_kinds, &icalset_fileset_init);
172 icalpvl_push(icalset_kinds, &icalset_dirset_init);
173#if defined(HAVE_BDB)
174 icalpvl_push(icalset_kinds, &icalset_bdb4set_init);
175#endif
176
177 icalset_init_done++;
178}

182 if (!icalset_init_done) {
183 icalset_init();
184 }

186 icalpvl_push(icalset_kinds, set);
187 return true;

192icalset *icalset_new(icalset_kind kind, const char *dsn, void *options)
193{
194 icalset *data = NULL;
195 icalset *ret = NULL;
196
197#if defined(_DLOPEN_TEST)
198 icalpvl_elem e;
199 const icalset *impl;
200
201 if (!icalset_init_done) {
202 icalset_init();
203 }
204
205 for (e = icalpvl_head(icalset_kinds); e != 0; e = icalpvl_next(e)) {
206 impl = (icalset *)icalpvl_data(e);
207 if (impl->kind == kind) {
208 break;
209 }
210 }
211 if (e == 0) {
212 icalerror_set_errno(ICAL_UNIMPLEMENTED_ERROR);
213 return (NULL);
214 }
215
216 data = (icalset *)malloc(impl->size);
217 if (data == (icalset *)NULL) {
218 icalerror_set_errno(ICAL_NEWFAILED_ERROR);
219 errno = ENOMEM;
220 return 0;
221 }
222
223 /* The first member of the derived class must be an icalset. */
224 memset(data, 0, impl->size);
225 /* *data = *impl; */
226 memcpy(data, impl, sizeof(icalset));
227
228 data->dsn = strdup(dsn);
229#else
230 switch (kind) {
231 case ICAL_FILE_SET: {
232 icalfileset *fdata;
233 fdata = (icalfileset *)malloc(sizeof(icalfileset));
234 data = (icalset *)fdata;
235 if (data == 0) {
236 icalerror_set_errno(ICAL_NEWFAILED_ERROR);
237 errno = ENOMEM;
238 return 0;
239 }
240 memset(data, 0, sizeof(icalfileset));
241 *data = icalset_fileset_init;
242 break;
243 }
244 case ICAL_DIR_SET: {
245 icaldirset *ddata;
246 ddata = (icaldirset *)malloc(sizeof(icaldirset));
247 data = (icalset *)ddata;
248 if (data == 0) {
249 icalerror_set_errno(ICAL_NEWFAILED_ERROR);
250 errno = ENOMEM;
251 return 0;
252 }
253 memset(data, 0, sizeof(icaldirset));
254 *data = icalset_dirset_init;
255 break;
256 }
257#if defined(HAVE_BDB)
258 case ICAL_BDB_SET: {
259 icalbdbset *bdata;
260 bdata = (icalbdbset *)malloc(sizeof(icalbdbset));
261 data = (icalset *)bdata;
262 if (data == 0) {
263 icalerror_set_errno(ICAL_NEWFAILED_ERROR);
264 errno = ENOMEM;
265 return 0;
266 }
267 memset(data, 0, sizeof(icalbdbset));
268 *data = icalset_bdbset_init;
269 break;
270 }
271#endif
272
273 default:
274 icalerror_set_errno(ICAL_UNIMPLEMENTED_ERROR);
276 return (NULL);
277 }
278
279 data->kind = kind;
280 data->dsn = strdup(dsn);
281#endif
282
284 if ((ret = data->init(data, dsn, options)) == NULL) {
285 /* coverity[alloc_free_mismatch] */
286 icalset_free(data);
287 }
288
289 return ret;
290}

420icalcomponent *icalsetiter_next(icalsetiter *itr)
421{
422 icalcomponent *c = 0;
423
424 icalerror_check_arg_rz((itr != NULL), "i");
425
426 do {
427 c = icalcompiter_next(&(itr->iter));
428 if (c != 0 && (itr->gauge == 0 || icalgauge_compare(itr->gauge, c) == 1)) {
429 return c;
430 }
431 } while (c != 0);
432
433 return 0;
434}
