25struct icalperiodtype icalperiodtype_from_string(const char *str)
26{
27 struct icalperiodtype p, null_p;
28 char *s = icalmemory_strdup(str);
29 char *start, *end;
30 icalerrorstate es;
31
32 /* Errors are normally generated in the following code, so save
33 the error state for resoration later */
34
35 icalerrorenum e = icalerrno;
36
37 p.start = p.end = icaltime_null_time();
38 p.duration = icaldurationtype_null_duration();
39
40 null_p = p;
41
42 if (s == 0) {
43 goto error;
44 }
45
46 start = s;
47 end = strchr(s, '/');
48
49 if (end == 0) {
50 goto error;
51 }
52
53 *end = 0;
54 end++;
55
56 p.start = icaltime_from_string(start);
57
58 if (icaltime_is_null_time(p.start)) {
59 goto error;
60 }
61
62 es = icalerror_get_error_state(ICAL_MALFORMEDDATA_ERROR);
63 icalerror_set_error_state(ICAL_MALFORMEDDATA_ERROR, ICAL_ERROR_NONFATAL);
64
65 p.end = icaltime_from_string(end);
66
67 icalerror_set_error_state(ICAL_MALFORMEDDATA_ERROR, es);
68
69 if (icaltime_is_null_time(p.end)) {
70 p.duration = icaldurationtype_from_string(end);
71
72 if (icaldurationtype_is_null_duration(p.duration)) {
73 goto error;
74 }
75 }
76
77 icalerrno = e;
78 icalmemory_free_buffer(s);
79 return p;
80
81error:
82 icalerror_set_errno(ICAL_MALFORMEDDATA_ERROR);
83 icalmemory_free_buffer(s);
84 return null_p;
85}

96char *icalperiodtype_as_ical_string_r(struct icalperiodtype p)
97{
98 const char *start;
99 const char *end;
100
101 char *buf;
102 size_t buf_size = 40;
103 char *buf_ptr = 0;
104
105 buf = (char *)icalmemory_new_buffer(buf_size);
106 buf_ptr = buf;
107
108 start = icaltime_as_ical_string_r(p.start);
109 icalmemory_append_string(&buf, &buf_ptr, &buf_size, start);
110 icalmemory_free_buffer((void *)start);
111
112 if (!icaltime_is_null_time(p.end)) {
113 end = icaltime_as_ical_string_r(p.end);
114 } else {
115 end = icaldurationtype_as_ical_string_r(p.duration);
116 }
117
118 icalmemory_append_char(&buf, &buf_ptr, &buf_size, '/');
119
120 icalmemory_append_string(&buf, &buf_ptr, &buf_size, end);
121 icalmemory_free_buffer((void *)end);
122
123 return buf;
124}
