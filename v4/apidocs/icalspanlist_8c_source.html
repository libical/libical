33static int compare_span(void *a, void *b)
34{
35 const struct icaltime_span *span_a = (struct icaltime_span *)a;
36 const struct icaltime_span *span_b = (struct icaltime_span *)b;
37
38 if (span_a->start == span_b->start) {
39 return 0;
40 } else if (span_a->start < span_b->start) {
41 return -1;
42 } else { /*if(span_a->start > span->b.start) */
43 return 1;
44 }
45}

58static void icalspanlist_new_callback(icalcomponent *comp, const struct icaltime_span *span, void *data)
59{
60 icaltime_span *s;
61 icalspanlist *sl = (icalspanlist *)data;
62
63 _unused(comp);
64
65 if (span->is_busy == 0) {
66 return;
67 }
68
69 if ((s = (icaltime_span *)malloc(sizeof(icaltime_span))) == 0) {
70 icalerror_set_errno(ICAL_NEWFAILED_ERROR);
71 return;
72 }
73
74 /* copy span data into allocated memory.. **/
75 *s = *span;
76 icalpvl_insert_ordered(sl->spans, compare_span, (void *)s);
77}

79icalspanlist *icalspanlist_new(icalset *set, struct icaltimetype start, struct icaltimetype end)
80{
81 struct icaltime_span range;
82 icalpvl_elem itr;
83 icalcomponent_kind inner_kind;
84 icalspanlist *sl;
85 struct icaltime_span *freetime;
86
87 if ((sl = (struct icalspanlist_impl *)malloc(sizeof(struct icalspanlist_impl))) == 0) {
88 icalerror_set_errno(ICAL_NEWFAILED_ERROR);
89 return 0;
90 }
91
92 sl->spans = icalpvl_newlist();
93 sl->start = start;
94 sl->end = end;
95
96 range.start = icaltime_as_timet(start);
97 range.end = icaltime_as_timet(end);
98
99 /* Gets a list of spans of busy time from the events in the set
100 and order the spans based on the start time */
101
102 for (icalcomponent *c = icalset_get_first_component(set);
103 c != 0;
104 c = icalset_get_next_component(set)) {
105 icalcomponent_kind kind = icalcomponent_isa(c);
106 icalcomponent *inner = icalcomponent_get_inner(c);
107
108 if (!inner) {
109 continue;
110 }
111
112 inner_kind = icalcomponent_isa(inner);
113
114 if (kind != ICAL_VEVENT_COMPONENT && inner_kind != ICAL_VEVENT_COMPONENT) {
115 continue;
116 }
117
118 icalerror_clear_errno();
119
120 icalcomponent_foreach_recurrence(c, start, end, icalspanlist_new_callback, (void *)sl);
121 }
122
123 /* Now Fill in the free time spans. loop through the spans. if the
124 start of the range is not within the span, create a free entry
125 that runs from the start of the range to the start of the
126 span. */
127
128 for (itr = icalpvl_head(sl->spans); itr != 0; itr = icalpvl_next(itr)) {
129 const struct icaltime_span *s = (struct icaltime_span *)icalpvl_data(itr);
130
131 if (!s) {
132 continue;
133 }
134
135 if ((freetime = (struct icaltime_span *)malloc(sizeof(struct icaltime_span))) == 0) {
136 icalerror_set_errno(ICAL_NEWFAILED_ERROR);
137 icalspanlist_free(sl);
138 return 0;
139 }
140
141 if (range.start < s->start) {
142 freetime->start = range.start;
143 freetime->end = s->start;
144
145 freetime->is_busy = 0;
146
147 icalpvl_insert_ordered(sl->spans, compare_span, (void *)freetime);
148 } else {
149 free(freetime);
150 }
151
152 range.start = s->end;
153 }
154
155 /* If the end of the range is null, then assume that everything
156 after the last item in the calendar is open and add a span
157 that indicates this */
158
159 if (icaltime_is_null_time(end)) {
160 const struct icaltime_span *last_span;
161
162 last_span = (struct icaltime_span *)icalpvl_data(icalpvl_tail(sl->spans));
163
164 if (last_span != 0) {
165 if ((freetime = (struct icaltime_span *)malloc(sizeof(struct icaltime_span))) == 0) {
166 icalerror_set_errno(ICAL_NEWFAILED_ERROR);
167 icalspanlist_free(sl);
168 return 0;
169 }
170
171 freetime->is_busy = 0;
172 freetime->start = last_span->end;
173 freetime->end = freetime->start;
174 icalpvl_insert_ordered(sl->spans, compare_span, (void *)freetime);
175 }
176 }
177
178 return sl;
179}

181void icalspanlist_free(icalspanlist *sl)
182{
183 struct icaltime_span *span;
184
185 if (sl == NULL) {
186 return;
187 }
188
189 while ((span = icalpvl_pop(sl->spans)) != 0) {
190 free(span);
191 }
192
193 icalpvl_free(sl->spans);
194
195 sl->spans = 0;
196
197 free(sl);
198}

214struct icalperiodtype icalspanlist_next_free_time(icalspanlist *sl, struct icaltimetype t)
215{
216 icalpvl_elem itr;
217 struct icalperiodtype period;
218 struct icaltime_span *s;
219
220 icaltime_t rangett = icaltime_as_timet(t);
221
222 period.start = icaltime_null_time();
223 period.end = icaltime_null_time();
224 period.duration = icaldurationtype_null_duration();
225
226 itr = icalpvl_head(sl->spans);
227 s = (struct icaltime_span *)icalpvl_data(itr);
228
229 if (s == 0) {
230 /* No elements in span */
231 return period;
232 }
233
234 /* Is the reference time before the first span? If so, assume
235 that the reference time is free */
236 if (rangett < s->start) {
237 /* End of period is start of first span if span is busy, end
238 of the span if it is free */
239 period.start = t;
240
241 if (s->is_busy == 1) {
242 period.end = icaltime_from_timet_with_zone(s->start, 0, NULL);
243 } else {
244 period.end = icaltime_from_timet_with_zone(s->end, 0, NULL);
245 }
246
247 return period;
248 }
249
250 /* Otherwise, find the first free span that contains the
251 reference time. */
252 for (itr = icalpvl_head(sl->spans); itr != 0; itr = icalpvl_next(itr)) {
253 s = (struct icaltime_span *)icalpvl_data(itr);
254
255 if (!s) {
256 continue;
257 }
258
259 if (s->is_busy == 0 && s->start >= rangett && (rangett < s->end || s->end == s->start)) {
260 if (rangett < s->start) {
261 period.start = icaltime_from_timet_with_zone(s->start, 0, NULL);
262 } else {
263 period.start = icaltime_from_timet_with_zone(rangett, 0, NULL);
264 }
265
266 period.end = icaltime_from_timet_with_zone(s->end, 0, NULL);
267
268 return period;
269 }
270 }
271
272 period.start = icaltime_null_time();
273 period.end = icaltime_null_time();
274
275 return period;
276}

278int *icalspanlist_as_freebusy_matrix(icalspanlist *spanlist, int delta_t)
279{
280 icalpvl_elem itr;
281 icaltime_t spanduration_secs;
282 int *matrix;
283 icaltime_t matrix_slots;
284 icaltime_t sl_start, sl_end;
285
286 icalerror_check_arg_rz((spanlist != 0), "spanlist");
287
288 if (!delta_t) {
289 delta_t = 3600;
290 }
291
292 /* calculate the start and end time as icaltime_t **/
293 sl_start = icaltime_as_timet_with_zone(spanlist->start, icaltimezone_get_utc_timezone());
294 sl_end = icaltime_as_timet_with_zone(spanlist->end, icaltimezone_get_utc_timezone());
295
296 /* insure that the time period falls on a time boundary divisible
297 by delta_t */
298
299 sl_start /= delta_t;
300 sl_start *= delta_t;
301
302 sl_end /= delta_t;
303 sl_end *= delta_t;
304
305 /* find the duration of this spanlist **/
306 spanduration_secs = sl_end - sl_start;
307
308 /* malloc our matrix, add one extra slot for a final -1 **/
309 matrix_slots = spanduration_secs / delta_t + 1;
310
311 matrix = (int *)malloc(sizeof(int) * (size_t)matrix_slots);
312 if (matrix == NULL) {
313 icalerror_set_errno(ICAL_NEWFAILED_ERROR);
314 return NULL;
315 }
316 memset(matrix, 0, sizeof(int) * (size_t)matrix_slots);
317 matrix[matrix_slots - 1] = -1;
318
319 /* loop through each span and mark the slots in the array */
320
321 for (itr = icalpvl_head(spanlist->spans); itr != 0; itr = icalpvl_next(itr)) {
322 const struct icaltime_span *s = (struct icaltime_span *)icalpvl_data(itr);
323
324 if (s && s->is_busy == 1) {
325 icaltime_t offset_start = s->start / delta_t - sl_start / delta_t;
326 icaltime_t offset_end = (s->end - 1) / delta_t - sl_start / delta_t + 1;
327 icaltime_t i;
328
329 if (offset_end >= matrix_slots) {
330 offset_end = matrix_slots - 1;
331 }
332
333 for (i = offset_start; i < offset_end; i++) {
334 matrix[i]++;
335 }
336 }
337 }
338 return matrix;
339}

341icalcomponent *icalspanlist_as_vfreebusy(icalspanlist *sl,
342 const char *organizer, const char *attendee)
343{
344 icalcomponent *comp;
345 struct icaltimetype atime = icaltime_from_timet_with_zone(time(0), 0, NULL);
346 icalpvl_elem itr;
347 icaltimezone *utc_zone;
348
349 if (!attendee) {
350 icalerror_set_errno(ICAL_USAGE_ERROR);
351 return 0;
352 }
353
354 utc_zone = icaltimezone_get_utc_timezone();
355
356 comp = icalcomponent_new_vfreebusy();
357
358 icalcomponent_add_property(comp, icalproperty_new_dtstart(sl->start));
359 icalcomponent_add_property(comp, icalproperty_new_dtend(sl->end));
360 icalcomponent_add_property(comp, icalproperty_new_dtstamp(atime));
361
362 if (organizer) {
363 icalcomponent_add_property(comp, icalproperty_new_organizer(organizer));
364 }
365 icalcomponent_add_property(comp, icalproperty_new_attendee(attendee));
366
367 /* now add the freebusy sections.. */
368
369 for (itr = icalpvl_head(sl->spans); itr != 0; itr = icalpvl_next(itr)) {
370 struct icaltime_span *s = (struct icaltime_span *)icalpvl_data(itr);
371
372 if (s && s->is_busy == 1) {
373 struct icalperiodtype period;
374 period.start = icaltime_from_timet_with_zone(s->start, 0, utc_zone);
375 period.end = icaltime_from_timet_with_zone(s->end, 0, utc_zone);
376 period.duration = icaldurationtype_null_duration();
377
378 icalproperty *p = icalproperty_new_freebusy(period);
379 icalproperty_add_parameter(p, icalparameter_new_fbtype(ICAL_FBTYPE_BUSY));
380 icalcomponent_add_property(comp, p);
381 }
382 }
383
384 return comp;
385}

387icalspanlist *icalspanlist_from_vfreebusy(icalcomponent *comp)
388{
389 icalcomponent *inner;
390 icalproperty *prop;
391 icalspanlist *sl;
392
393 icalerror_check_arg_rz((comp != NULL), "comp");
394
395 inner = icalcomponent_get_inner(comp);
396 if (!inner) {
397 return NULL;
398 }
399
400 if ((sl = (icalspanlist *)malloc(sizeof(icalspanlist))) == 0) {
401 icalerror_set_errno(ICAL_NEWFAILED_ERROR);
402 return 0;
403 }
404 sl->spans = icalpvl_newlist();
405
406 /* cycle through each FREEBUSY property, adding to the spanlist */
407 for (prop = icalcomponent_get_first_property(inner, ICAL_FREEBUSY_PROPERTY);
408 prop != NULL;
409 prop = icalcomponent_get_next_property(inner, ICAL_FREEBUSY_PROPERTY)) {
410 icaltime_span *s = (icaltime_span *)malloc(sizeof(icaltime_span));
411 icalparameter *param;
412 struct icalperiodtype period;
413 icalparameter_fbtype fbtype;
414
415 if (s == 0) {
416 icalerror_set_errno(ICAL_NEWFAILED_ERROR);
417 icalspanlist_free(sl);
418 return 0;
419 }
420
421 param = icalproperty_get_first_parameter(prop, ICAL_FBTYPE_PARAMETER);
422 fbtype = (param) ? icalparameter_get_fbtype(param) : ICAL_FBTYPE_BUSY;
423
424 switch (fbtype) {
425 case ICAL_FBTYPE_FREE:
426 case ICAL_FBTYPE_NONE:
427 case ICAL_FBTYPE_X:
428 s->is_busy = 1;
429 break;
430 default:
431 s->is_busy = 0;
432 }
433
434 period = icalproperty_get_freebusy(prop);
435 s->start = icaltime_as_timet_with_zone(period.start, icaltimezone_get_utc_timezone());
436 s->end = icaltime_as_timet_with_zone(period.end, icaltimezone_get_utc_timezone());
437
438 icalpvl_insert_ordered(sl->spans, compare_span, (void *)s);
439 }
441 return sl;
442}
