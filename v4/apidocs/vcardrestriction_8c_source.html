81int vcardrestriction_compare(vcardrestriction_kind restr, int count)
82{
83 /* restr is an unsigned int and VCARD_RESTRICTION_NONE == 0,
84 so no need to check if restr < VCARD_RESTRICTION_NONE */
85 if (restr > VCARD_RESTRICTION_UNKNOWN || count < 0) {
86 return -1;
87 }
88
89 if (count > 2) {
90 count = 2;
91 }
92
93 return compare_map[restr][count];
94}

100static const char *vcardrestriction_validate_datetime_value(
101 const vcardrestriction_record *rec,
102 vcardcomponent *comp, vcardproperty *prop)
103{
104 vcardtimetype t = vcardproperty_get_bday(prop);
105 static char buf[TMP_BUF_SIZE];
106
107 if (vcardtime_is_null_datetime(t)) {
108 return 0;
109 }
110
111 if (comp && vcardcomponent_get_version(comp) != VCARD_VERSION_40) {
112 unsigned missing_time_parts = (unsigned)((t.hour < 0) + (t.minute < 0) + (t.second < 0));
113
114 if (t.year < 0 || t.month < 0 || t.day < 0 ||
115 (missing_time_parts && missing_time_parts != 3)) {
116
117 snprintf(buf, TMP_BUF_SIZE,
118 "Failed restrictions for %s property. "
119 "The value must be a full date or date-time",
120 vcardproperty_kind_to_string(rec->property));
121 return buf;
122 }
123 }
124
125 if (!vcardtime_is_valid_time(t)) {
126 snprintf(buf, TMP_BUF_SIZE,
127 "Failed restrictions for %s property. "
128 "The value is an invalid date-and-or-time",
129 vcardproperty_kind_to_string(rec->property));
130 return buf;
131 }
132
133 return 0;
134}

136static const char *vcardrestriction_validate_timestamp_value(
137 const vcardrestriction_record *rec,
138 vcardcomponent *comp, vcardproperty *prop)
139{
140 _unused(comp);
141
142 vcardtimetype t = vcardproperty_get_rev(prop);
143 static char buf[TMP_BUF_SIZE];
144
145 if (vcardtime_is_null_datetime(t)) {
146 return 0;
147 }
148
149 if (!vcardtime_is_timestamp(t) || !vcardtime_is_valid_time(t)) {
150 snprintf(buf, TMP_BUF_SIZE,
151 "Failed restrictions for %s property. "
152 "The value is an invalid timestamp",
153 vcardproperty_kind_to_string(rec->property));
154 return buf;
155 }
156
157 return 0;
158}

160static int _check_restriction(vcardcomponent *comp,
161 const vcardrestriction_record *record,
162 int count, vcardproperty *prop)
163{
164 vcardrestriction_kind restr;
165 const char *funcr = 0;
166 int compare;
167
168 restr = record->restriction;
169
170 if (restr == VCARD_RESTRICTION_ONEEXCLUSIVE ||
171 restr == VCARD_RESTRICTION_ONEMUTUAL) {
172
173 /* First treat it as a 0/1 restriction */
174 restr = VCARD_RESTRICTION_ZEROORONE;
175 }
176
177 compare = vcardrestriction_compare(restr, count);
178
179 assert(compare != -1);
180
181 if (compare == 0) {
182 char temp[TMP_BUF_SIZE];
183 vcardproperty *errProp;
184 vcardparameter *errParam;
185 const char *type, *kind;
186
187 if (record->subcomponent != VCARD_NO_COMPONENT) {
188 type = "component";
189 kind = vcardcomponent_kind_to_string(record->subcomponent);
190 } else {
191 type = "property";
192 kind = vcardproperty_kind_to_string(record->property);
193 }
194
195 snprintf(temp, TMP_BUF_SIZE,
196 "Failed restrictions for %s %s. "
197 "Expected %s instances of the %s and got %d",
198 kind, type, restr_string_map[restr], type, count);
199 errParam =
200 vcardparameter_new_xlicerrortype(VCARD_XLICERRORTYPE_RESTRICTIONCHECK);
201 errProp = vcardproperty_vanew_xlicerror(temp, errParam, (void *) 0);
202 vcardcomponent_add_property(comp, errProp);
203 vcardproperty_free(errProp);
204 }
205
206 if (record->function != NULL) {
207 funcr = record->function(record, comp, prop);
208 }
209
210 if (funcr != 0) {
211 vcardproperty *errProp;
212 vcardparameter *errParam;
213
214 errParam =
215 vcardparameter_new_xlicerrortype(VCARD_XLICERRORTYPE_RESTRICTIONCHECK);
216 errProp = vcardproperty_vanew_xlicerror(funcr, errParam, (void *) 0);
217 vcardcomponent_add_property(comp, errProp);
218 vcardproperty_free(errProp);
219
220 compare = 0;
221 }
222
223 return compare;
224}

226static bool vcardrestriction_check_component(vcardcomponent *comp)
227{
228 vcardcomponent_kind comp_kind, inner_kind;
229 vcardproperty_version version;
230 vcardproperty_kind prop_kind;
231 const vcardrestriction_record *start_record;
232 vcardproperty *version_prop = NULL;
233 vcardcomponent *inner_comp;
234 int count;
235 int compare;
236 bool valid = true;
237
238 comp_kind = vcardcomponent_isa(comp);
239
240 switch (comp_kind) {
241 case VCARD_VCARD_COMPONENT:
242 /* Get the VERSION property from the component */
243 version_prop =
244 vcardcomponent_get_first_property(comp, VCARD_VERSION_PROPERTY);
245 break;
246
247 default:
248 break;
249 }
250
251 if (version_prop == 0) {
252 version = VCARD_VERSION_NONE;
253 } else {
254 version = vcardproperty_get_version(version_prop);
255 }
256
257 /* Check all of the properties in this component */
258
259 start_record = vcardrestriction_get_restriction(NULL, VCARD_VERSION_NONE,
260 comp_kind,
261 VCARD_ANY_PROPERTY,
262 VCARD_NO_COMPONENT);
263 if (start_record != &null_restriction_record) {
264
265 for (prop_kind = VCARD_ANY_PROPERTY + 1;
266 prop_kind != VCARD_NO_PROPERTY; prop_kind++) {
267
268 const vcardrestriction_record *record =
269 vcardrestriction_get_restriction(start_record, version, comp_kind,
270 prop_kind, VCARD_NO_COMPONENT);
271
272 vcardproperty *prop =
273 vcardcomponent_get_first_property(comp, prop_kind);
274
275 count = vcardcomponent_count_properties(comp, prop_kind, 1);
276
277 compare = _check_restriction(comp, record, count, prop);
278
279 valid = valid && compare;
280 }
281 }
282
283 /* Now check the inner components */
284
285 start_record = vcardrestriction_get_restriction(NULL, VCARD_VERSION_NONE,
286 comp_kind,
287 VCARD_NO_PROPERTY,
288 VCARD_ANY_COMPONENT);
289
290 if (start_record != &null_restriction_record) {
291
292 for (inner_kind = VCARD_NO_COMPONENT + 3;
293 inner_kind != VCARD_NUM_COMPONENT_TYPES; inner_kind++) {
294
295 const vcardrestriction_record *record =
296 vcardrestriction_get_restriction(start_record, version, comp_kind,
297 VCARD_NO_PROPERTY, inner_kind);
298
299 count = vcardcomponent_count_components(comp, inner_kind);
300
301 compare = _check_restriction(comp, record, count, NULL);
302
303 valid = valid && compare;
304 }
305 }
306
307 for (inner_comp = vcardcomponent_get_first_component(comp, VCARD_ANY_COMPONENT);
308 inner_comp != 0;
309 inner_comp = vcardcomponent_get_next_component(comp, VCARD_ANY_COMPONENT)) {
310
311 compare = vcardrestriction_check_component(inner_comp);
312
313 valid = valid && compare;
314 }
315
316 return valid;
317}

319bool vcardrestriction_check(vcardcomponent *outer_comp)
320{
321 vcardcomponent_kind comp_kind;
322 bool valid;
323
324 icalerror_check_arg_rz((outer_comp != 0), "outer comp");
325
326 comp_kind = vcardcomponent_isa(outer_comp);
327
328 if (comp_kind != VCARD_VCARD_COMPONENT &&
329 comp_kind != VCARD_XROOT_COMPONENT) {
330 icalerror_set_errno(ICAL_BADARG_ERROR);
331 return 0;
332 }
333
334 valid = vcardrestriction_check_component(outer_comp);
335
336 return valid;
337}

363static const vcardrestriction_record *vcardrestriction_get_restriction(
364 const vcardrestriction_record *start,
365 vcardproperty_version version, vcardcomponent_kind component,
366 vcardproperty_kind property, vcardcomponent_kind subcomp)
367{
368 const vcardrestriction_record *rec;
369
370 if (!start) {
371 start = &vcardrestriction_records[0];
372 }
373
374 for (rec = start; rec && rec->restriction != VCARD_RESTRICTION_NONE; rec++) {
375
376 if ((version == VCARD_VERSION_NONE ||
377 rec->version == VCARD_VERSION_NONE || version == rec->version) &&
378 (component == VCARD_ANY_COMPONENT ||
379 (component == rec->component &&
380 (property == VCARD_ANY_PROPERTY || property == rec->property) &&
381 (subcomp == VCARD_ANY_COMPONENT || subcomp == rec->subcomponent)))) {
382 return rec;
383 }
384 }
385
386 return &null_restriction_record;
